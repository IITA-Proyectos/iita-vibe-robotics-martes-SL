from pybricks.hubs import PrimeHub

from pybricks.hubs import PrimeHub
from pybricks.pupdevices import Motor
from pybricks.parameters import Port, Direction
from pybricks.robotics import DriveBase
from pybricks.tools import wait

hub = PrimeHub()

# 🔥 calibración inicial del gyro (quieto)

hub.imu.reset_heading(0)

left_motor = Motor(Port.A)
right_motor = Motor(Port.B, Direction.COUNTERCLOCKWISE)

wheel_diameter = 56
axle_track = 120

robot = DriveBase(left_motor, right_motor, wheel_diameter, axle_track)

robot.settings(straight_speed=200, turn_rate=90)

DISTANCIA_TOTAL = 220
GIRO = 98

# 🔥 alineación inicial (una sola vez)
def alinear_inicio():
    while abs(hub.imu.heading()) > 1:
        if hub.imu.heading() > 0:
            robot.drive(0, -30)
        else:
            robot.drive(0, 30)
        wait(10)
    robot.stop()

# 🔥 avance progresivo SIN pausas
def avanzar_pro():

    while distancia < DISTANCIA_TOTAL:

        if distancia < 50:
            velocidad = 100
        elif distancia < 170:
            velocidad = 300
        else:
            velocidad = 100

        robot.drive(velocidad, 0)
        wait(10)

        distancia += velocidad * 0.01  # aproximación

    robot.stop()
